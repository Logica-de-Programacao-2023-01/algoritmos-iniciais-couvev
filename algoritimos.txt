qnt = int(input("Quantos numero voce vai usar? "))
lst = []

for a in range(qnt):
      
      num = float(input("Numero: "))
      lst.append(num)
      sumlst = sum(lst)
      lenlst = len(lst)
      resu = sumlst/lenlst
      
print("A media dos seus numeros é: ",resu)
